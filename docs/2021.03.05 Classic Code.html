import matplotlib.pyplot as plt
from matplotlib.ticker import StrMethodFormatter as SMF
import math
import numpy as np
import pandas as pd

#Final data of games, hits, and rounded/exact
bat_avg = pd.DataFrame(columns =["Games",
                                 "Hits",
                                 "Error"])

for game in range(1, 1000):
    
    # hits/4g ≈ g/1000
    hits = game*game / 250.0

    b = math.floor(hits)
    bottom = round(b / (4 * game), 3)
    t = math.ceil(hits)
    top = round(t / (4 * game), 3)

    #Save if equality (multiples of 50).
    if (b == t):
        bat_avg = bat_avg.append({"Games": game,
                                  "Hits": b,
                                  "Error": "exact"},
                                 ignore_index = True)

    #Save if batting average rounds down to game/1000.
    elif (game/1000 == bottom):
        bat_avg = bat_avg.append({"Games": game,
                                  "Hits": b,
                                  "Error": "rounded"},
                                 ignore_index = True)
        
    #Save if batting average rounds up to game/1000.
    elif (game/1000 == top):
        bat_avg = bat_avg.append({"Games": game,
                                  "Hits": t,
                                  "Error": "rounded"},
                                 ignore_index = True)

#Rounded and exact.
bat_avg.set_index('Games', inplace = True)
rounded = bat_avg[bat_avg['Error'] == 'rounded']
exact = bat_avg[bat_avg['Error'] == 'exact']

#Scatter plot helper.
def scatter_helper(axes, df, color):
    ax.scatter(x = df.index.values,
               y = df['Hits'],
               lw = 4,
               c = color)

    
#Plot.    
fig = plt.figure(figsize = (13, 8))
ax = fig.add_subplot()

#Rounded and exact in different colors.
scatter_helper(ax, rounded, '#1F968B')
scatter_helper(ax, exact, '#481567')

ax.set_title("# Hits for Matching Rounded Batting Average", fontsize = 24)

ax.set_xlabel("Game Number", fontsize = 18)
plt.xticks(rotation = 25)

ax.set_ylabel("Total Number of Hits", fontsize = 18)
ax.yaxis.set_major_formatter(SMF('{x: ,.0f}'))
plt.yticks(rotation = 25)

ax.tick_params(axis = 'both', labelsize = 16)
