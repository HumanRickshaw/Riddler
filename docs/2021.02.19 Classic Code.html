top = []

for i in range(50000000):
    
    #Current block.
    block = 1
    #Current blocks.
    blocks = {1:0}
    #Stability.
    stable = True
    while stable :
    
        #Center of current block, b.
        loc = blocks[block]
        #Add a new block, b+1.
        block += 1
        blocks[block] = ru(loc-1, loc+1)
    
        for k in range(2, block):
            #Determine the center of the top b blocks, blocks b-k+2 to b+1.
            centers = list(blocks.values())[block-k : block+1]
            avg_center = sum(centers) / k
            #Grab the center of the b-k-1 block.
            base_center = blocks[block - k]
            #Check if the average center is outside the base center. 
            if ((avg_center < base_center - 1) or (base_center + 1 < avg_center)):
                top.append(block)
                stable = False
                break

print(sum(top) / 50000000)

#Plot Helper.
def tower_plot(x, y, index):
    
    t = ["", "Cumulative "][index]
    ylabel = ["Number", "Percent"][index]
    c = ["#440154FF", "#21908CFF"][index]
    
    sns.set()
    fig, ax = plt.subplots(figsize=(15, 9))
    ax.scatter(x = x,   
               y = y,
               s = 60,
               c = c)

    plt.suptitle(t + "Distribution of Maximum Height of Riddler Jenga", fontsize = 26)
    plt.title("50,000,000 Simulations", fontsize = 22)

    ax.set_xlabel("Height of Tower (Before Collapse)", fontsize = 22)
    plt.xticks(ticks = [5*x for x in range(14)])

    ax.set_ylabel(ylabel + " of Simulations", fontsize = 22)
    
    if (index == 0):
        plt.yticks(ticks = [1000000*y for y in range(8)], rotation = 25)
        ax.yaxis.set_major_formatter(SMF('{x: ,.0f}'))
    else:
        ax.yaxis.set_major_formatter(FMF('%.0f%%'))

    ax.tick_params(axis = 'both', labelsize = 16)
