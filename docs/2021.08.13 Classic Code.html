#Create a hand of n unfrozen dice.
def initial(n):
    dice = []
    for d in range(n):
        dice.append(random.randrange(1,7))
    dice.sort(reverse = True)
    return(dice)


#Returns the frozen dice. 
def freeze_n_roll(n, dice):
    #Freeze the maximum, one at a time.
    frozen_dice = []
    #Grab the largest from the dice.
    frozen_dice.append(dice[0])
    dice = dice[1:]
    n = len(dice)
    #That could be the only die.
    if (n == 0):
        return(frozen_dice)
    
    keep = []
    #This loop calculates the sum of the largest dice and the expected value of rerolling the smaller dice,
    #and compares it to expected value of rerolling all dice.  If at least one is comparison is true, then
    #the list of dice is passed through the algorithm.
    for i in range(n):
        keep.append(sum(dice[:(n-i)]) + eV[i] > eV[n])
    
    if bool(sum(keep)):
        frozen_dice += freeze_n_roll(n, dice)
        return(frozen_dice)
    
    #Reroll all dice has the greatest expected sum.  Do it.
    else:
        frozen_dice += freeze_n_roll(n, initial(n))
        return(frozen_dice)


def dice(n) :

    final_sum = sum(freeze_n_roll(n, initial(n)))
    #Final
    return(final_sum)

#Average of 10,000,000 simulations.
mean = round(statistics.mean(dice_totals), 5)

#Plot.
sns.set()
fig = plt.figure(figsize = (14, 8))
ax = fig.add_subplot()

#Histogram.
plt.hist(x = dice_totals,
         #Number of bins.
         bins = np.arange(len(set(dice_totals)) + 1) + min(set(dice_totals)) - 0.5,
         align = "mid",
         color = '#1F968B')
#Average value vertical line.
