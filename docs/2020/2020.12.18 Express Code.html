import math
import matplotlib.pyplot as plt

#Constants
c1 = 1 / math.sqrt(2)
c2 = math.sqrt(3 / 8)

t1 = (7 - math.sqrt(15)) / math.sqrt(128)
t2 = (7 + math.sqrt(15)) / math.sqrt(128)

t3 = c2 - math.sqrt(1 / 8)
t4 = c2 + math.sqrt(1 / 8)

#Graph helper
def glass_disk(axis, title, c, outline) :
    
    #Remove axes and labels.
    for side in ['bottom', 'left', 'top', 'right']:
        axis.spines[side].set_visible(False)
    axis.set_xticks([])
    axis.set_yticks([])
    axis.set(xlim = (0, 1.3), ylim = (0, 1.3))
    
    #Glass.
    glass = plt.Circle((0, 0), 1, color = 'lightblue', zorder = 1)
    axis.add_artist(glass)
    
    #Disk.
    disk = plt.Circle((c, c), 0.5, color = 'grey', alpha = 0.5, zorder = 2)
    axis.add_artist(disk)
    axis.scatter(c, c, color = "black", s = 100, zorder = 3)
    
    
    #Triangle.
    axis.plot(outline[0], outline[1], color = "blue", lw = 1, ls = ":", zorder = 3)
    
    #Title.
    axis.set_title(title, fontsize = 24)
