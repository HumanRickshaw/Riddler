#Function to draw cookie solutions.
def cookies_in_dough(figsize, xlim, ylim, dough, outline, cookie1, cookie2, cookie3):
    
    fig = plt.figure(figsize = figsize)
    ax = fig.add_subplot(111, xlim = xlim, ylim = ylim)

    #Remove axes and ticks.
    for side in ['bottom', 'left', 'top', 'right']:
        ax.spines[side].set_visible(False)
    ax.set_xticks([])
    ax.set_yticks([])

    ax.fill(dough[0], dough[1], c = 'yellow')
    ax.plot(outline[0], outline[1], c = 'black', ls = ':')

    for cookie in [cookie1, cookie2, cookie3]:
        ax.fill(cookie[0], cookie[1], c = 'orange')
        ax.plot(cookie[0], cookie[1], c = 'black', ls = '-', lw = 3)

#Outline of dough.
dough = [(-0.5, -0.5, 0.5, -0.5),
         (-0.5, 0.5, -0.5, -0.5)]

#cos 22.5°.
x = math.sqrt(2 + math.sqrt(2)) / 2
#sin 22.5°.
y = math.sqrt(2 - math.sqrt(2)) / 2

#Various points for graphing
a = 1 / 2 - 1 / math.sqrt(2) 
b = (x - 1) / 2
n = (b - a) / 2

c = a - n / y

d = -1 * n * math.sqrt(2)
e = n * math.sqrt(2)

f = n / (1 - math.sqrt(2))
g = -1 * n

h = -1 / 2 - f
i = b - n

#Cookie coordinates.
cookie1 = [(0.0, a, c, -0.5, -0.5, 0.0),
           (0.0, -0.5, -0.5, c, a, 0.0)]

cookie2 = [(-0.5, d, f, h, -0.5, -0.5),
           ( 0.5, e, g, i, b, 0.5)]

cookie3 = [( 0.5, b, i, g, e, 0.5),
           (-0.5, -0.5, h, f, d, -0.5)]

#Graph.
cookies_in_dough((8, 8),
                 (-0.51, 0.51),
                 (-0.51, 0.51),
                 dough,
                 dough,
                 cookie1,
                 cookie2,
                 cookie3)
