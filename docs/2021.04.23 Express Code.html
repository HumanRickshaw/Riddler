    ax.spines[side].set_visible(False)
ax.set_xticks([])
ax.set_yticks([])
ax.set(xlim = (-5, 5), ylim = (-5, 5))

y = r*np.sin(n * math.pi/4 + math.pi/8)

#1/4 Dodecagon

n = np.arange(0, 3, 1)
x2 = r*np.cos(n * math.pi/6 + math.pi/12)
