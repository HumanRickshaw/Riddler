#Draw close up for isoceles.
def triAngle(g, h, i, num):
    
    fig = plt.figure(figsize = (9, 7.5))
    ax = fig.add_subplot(111, xlim = (-0.1, 1.1), ylim = (-0.1, 0.9))

    #Remove axes and ticks.
    for side in ['bottom', 'left', 'top', 'right']:
        ax.spines[side].set_visible(False)
    ax.set_xticks([])
    ax.set_yticks([])

    st = math.sqrt(3)
    
    #Triangle.
    ax.plot((0, 1, 0.5, 0),
            (0, 0, st/2, 0),
             c = 'k',
             ls = '-',
             lw = 2)
    #Center, D, E, F.
    ax.scatter((0.25, 0.5, 0.5, 0.75),
               (st/4, 0, st/6, st/4),
               c = 'k',
               ls = '-',
               lw = 3,
               zorder = 2)
    #G.
    ax.scatter(g[0], g[1],
               c = 'b',
               ls = '-',
               lw = 3,
               zorder = 2)
    #H.
    ax.scatter(h[0], h[1],
               c = 'r',
               ls = '-',
               lw = 3,
               zorder = 2)
    
    #I.
    ax.scatter(i[0], i[1],
               c = 'g',
               ls = '-',
               lw = 3,
               zorder = 2)
    
    variables = [("A", (1.02, -0.02), 'k'), 
                 ("B", (-0.05, -0.02), 'k'),
                 ("C", (0.52, 0.85), 'k'),
                 
                 ("D", (0.49, -0.05), 'k'), 
                 ("E", (0.20, st/4), 'k'),
                 ("F", (0.77, st/4), 'k'),
                                                   
                 ("G", (g[0], g[1]-0.05), 'b'),
                 ("H", (h[0]-0.05, h[1]-0.02), 'r'),
                 ("I", (i[0]+0.02, i[1]-0.02), 'g')]
    
    #Variables.
    for v in variables :
        plt.annotate(v[0], v[1], c = v[2], fontsize = 16)
 
    #GHI.
    ax.plot((g[0], h[0], i[0], g[0]),
            (g[1], h[1], i[1], g[1]),
            c = 'grey',
            ls = ':',
            lw = 2,
            zorder = 1)
    
    fig.savefig("2021.10.22 Classic Code" + str(num) + ".png")
