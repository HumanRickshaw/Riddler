B = []
C = []
mediocre = []

#Exhaustive approach.
#B and C can both pick 0 - 10.  Let us look at all 121 possibilities.
for b in range(0,11):
    for c in range(0,11):
        
        minigame = {"A" : 3,
                    "B" : b,
                    "C" : c}
        
        finalscore = {"A" : 6,
                      "B" : 8,
                      "C" : 10}
        
        cards = list(minigame.values())
        
        med = np.median(cards)
        split = cards.count(med)
        winnings =  med / split
        
        #Update winnings for each player.
        for player in ("A", "B", "C") :
            
            #The winner in each mini game had picked the median value.
            if minigame[player] == med :
                
                #The amount they receive is the median value divided by the number of players who picked it.
                finalscore[player] = finalscore[player] + winnings
                
        med_final = np.median(list(finalscore.values()))
        
        winners = ''
        #Update dataframe
        for player in ("A", "B", "C") :
            
            
            #The player with that median final score...
            if finalscore[player] == med_final :
                winners = winners+player
                
                #Won that minigame out of the 121.
        B.append(b)
        C.append(c)
        mediocre.append(winners)

outcomes = pd.DataFrame(data = {"B": B,
                                "C": C,
                                "Mediocre": mediocre})

#Score the 5th round.
def score(row) :
    #A, B, C tie.
    if row['B'] == 3 and row['C'] == 3:
         return "(1,1,1)"
    #A, B tie.    
    elif row['B'] == 3:
         return "(1.5,1.5,0)"
    #A, C tie.  
    elif row['C'] == 3:
         return "(1.5,0,1.5)"
    #B, C tie.  
    elif row['B'] == row['C']:
        if row['B']/2 == int(row['B']/2):
            return "(0," + str(int(row['B']/2)) + "," + str(int(row['B']/2)) + ")"
        else :
            return "(0," + str(row['B']/2) + "," + str(row['B']/2) + ")"
    #One clear median.
    else :
        med = np.median((3, row['B'], row['C']))
        if med == 3 :
            return "(3,0,0)"
        if med == row['B'] :
            return "(0," + str(row['B']) + ",0)"
        else :
            return "(0,0," + str(row['C']) + ")"    

#Group 5th round totals.
def color_letter(row) :
    if row['Mediocre'] == "A" :
        return '#DCE319'
    elif row['Mediocre'] == "BC" :
        return '#29AF7F'
    elif row['B'] < 3 and row['C'] < 3 :
        return '#238A8D'
    elif row['B'] == 3:
        return '#440154'
    elif row['C'] == 3:
        return '#453781'
    elif row['Mediocre'] == "C" :
        return '#73D055'
    elif row['Mediocre'] == "B" :
        return '#33638D'

outcomes['Score'] = outcomes.apply (lambda row: score(row), axis = 1)    
outcomes["Color"] = outcomes.apply (lambda row: color_letter(row), axis = 1)

def helper121(option) :
    fig = plt.figure(figsize = (12, 12))
    ax = fig.add_subplot(111, xlim = (-0.5, 10.5), ylim = (-0.5, 10.5))

    for i in range(len(outcomes)) :
        
        if option == "Score" :
            label = outcomes["Score"][i]
            color = "black"
            fw = "normal"
            size = 12
        else :
            label = outcomes["Mediocre"][i]
            color = outcomes["Color"][i]
            fw = "bold"
            size = 24
        
        plt.text(x = outcomes["B"][i],
                 y = outcomes["C"][i],
                 s = label,
                 ha = 'center',
                 va = 'center',
                 c = color,
                 fontweight = fw,
                 size = size)

    #Title setup.
    if option == "Score" :
        ax.set_title('5th Round Winnings', fontsize = 24)
    else :    
        ax.set_title('Final Winner after 5th Round', fontsize = 24)

    #X-axis setup.
    ax.set_xlabel("B's Choice", fontsize = 24)
    ax.set_xticks(range(11))

    #Y-axis setup.
    ax.set_ylabel("C's Choice", fontsize = 24)
    ax.set_yticks(range(11))

    ax.tick_params(axis = 'both', which = 'major', labelsize = 22)

    fig.savefig("2022.04.08 Classic " + option + ".png")
