leaves = []
dates = []
maxl = 0
for t in range(10000) :
    s = t/10000
    l = (s - 1) * math.log(1 - s)
    date = equinox + TD(seconds = seconds * s)
    leaves.append(l)
    dates.append(date)
    if l > maxl:
        maxl = l
        maxd = date

print(maxl, maxd)
