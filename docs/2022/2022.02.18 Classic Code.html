#Get a list of c+1 fractions of possiblities for c coins being flipped.
def coinDist(c) :
    #Denominator
    d = 2**c - 1
    dist = []
    for h in range(c+1) :
        #Binomial expansion coefficients for (h+t)^c
        dist.append(Fct(c)/(Fct(c-h) * Fct(h)) / d)
    return(dist)

#List of c+1 probabilities, representing the probability of not finding the lucky coin,
#from 0 to c coins.
def UpAllNightToGetLucky(c) :
    
    #0 coins results in never finding a lucky coin.
    #1 coins results in always finding a lucky coin.
    #2 coins results in 1/3 never finding a lucky coin.
    probs = [0, 1, 2/3] 
    
    #Loop from 0 to c coins.   
    for n in range(3, c+1):

        dist = coinDist(n)
        new_prob = 0
        
        #Loop from 0 to n-1 heads. 
        for h in range(n):
            #Probability of being unlucky from h coins.
            new_prob += probs[h] * dist[h]
        
        probs.append(new_prob)
        
    return(probs)
