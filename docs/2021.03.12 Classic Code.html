import math
import matplotlib
from matplotlib.animation import FuncAnimation as FA
from matplotlib.patches import Ellipse as mpE
from matplotlib.patches import Rectangle as mpR
import matplotlib.pyplot as plt
from matplotlib.ticker import FormatStrFormatter as FMF
import seaborn as sns

sns.set()

#Base as a proportion of Area.
radii = [r / 10000 for r in range(1, 3990)]

bases = []
laterals = []
heights = []
volumes = []

#Volume from Base.
for r in radii:
    
    b = r**2 * math.pi
    bases.append(b)
    l = 1 - 2*b
    laterals.append(l)
    h = l / (2 * math.sqrt(math.pi * b))
    heights.append(h)
    v = b * h
    volumes.append(v)



#Plot.    
fig = plt.figure(figsize = (13, 8))
ax = fig.add_subplot()

#Rounded and exact in different colors.

matplotlib.rc_file_defaults()

fig = plt.figure(figsize = (10, 8))
ax = fig.add_subplot(xlim = (-1, 99),
                     ylim = (-1, 79))

ax.set_title("Volume of Pie from Fixed Area Sheet of Crust", fontsize = 24)

#Remove axes and ticks.
ax.spines["top"].set_visible(False)
ax.spines["right"].set_visible(False)
ax.spines["bottom"].set_visible(False)
ax.spines["left"].set_visible(False)
ax.set_xticks([])
ax.set_yticks([])

font = {'size' : 16}

#Seven artists:
#Bottom, cylinder, bottom outline, and top.
#Base text, lateral text, volume text.
b1 = mpE(xy = (0, 0),
         width = 0,
         height = 0,
         edgecolor = 'goldenrod',
         ls = '-',
         lw = 6,
         facecolor = 'bisque')

c = mpR(xy = (0, 0),
        width = 0,
        height = 0,
        edgecolor = 'goldenrod',
        ls = '-',
        lw = 3,
        facecolor = 'bisque')

b2 = mpE(xy = (0, 0),
         width = 0,
         height = 0,
         edgecolor = 'goldenrod',
         ls = ':',
         lw = 2,
         facecolor = 'bisque')

t1 = mpE(xy = (0, 0),
         width = 0,
         height = 0,
         edgecolor = 'goldenrod',
         ls = '-',
         lw = 3,
         facecolor = 'bisque')

b_text = ax.text(45, 55, '', **font)
l_text = ax.text(45, 50, '', **font)
v_text = ax.text(45, 45, '', **font)


def init():
    """Initialize seven artists."""
    ax.add_patch(b1)
    ax.add_patch(c)
    ax.add_patch(b2)
    ax.add_patch(t1)

    b_text.set_text('')
    l_text.set_text('')
    v_text.set_text('')
    return b1, c, b2, t1, b_text, l_text, v_text

def animate(i):
    
    r = 100 * radii[i]
    h = 100 * heights[i]
    """Update seven artists."""
    b1.set_center([r, r/4])
    b1.set_width(2*r)
    b1.set_height(r/2)
    
    c.set_xy([-0.2, r/4])
    c.set_width(2*r + 0.4)
    c.set_height(h)
    
    b2.set_center([r, r/4])
    b2.set_width(2*r)
    b2.set_height(r/2)
    
    t1.set_center([r, r/4 + h])
    t1.set_width(2*r)
    t1.set_height(r/2)   

    #r_text.set_text("Radius: %.7f" % radii[i])
    #h_text.set_text("Height: %.7f" % heights[i])
    b_text.set_text("Base Area: %.2f%% of Sheet of Crust" % (100*bases[i]))
    l_text.set_text("Lateral Area: %.2f%% of Sheet of Crust" % (100*laterals[i]))
    v_text.set_text("Volume: %.7f" % volumes[i])

    return b1, c, b2, t1, b_text, l_text, v_text

#Run animation.
anim = FA(fig, animate, init_func = init, frames = 3989, interval = 5, blit = True)       

#Save animation.
anim.save('2021.03.12 maximum_pie.mp4');
