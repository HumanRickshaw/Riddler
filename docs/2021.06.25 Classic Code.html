#Initiate expected value, n, and k lists.
evs = []
ks = []
ns = []

for n in range(1, 13) :
    for k in range (n+1):
        ev = k * (n - k) / 2
        evs.append(ev)
        ks.append(k)
        ns.append(n)
        
riddlerSolitaire = pd.DataFrame(data = {"Expected Value": evs,
                                        "Cards Flipped": ks,
                                        "Cards in Deck": ns})
riddlerSolitaire
