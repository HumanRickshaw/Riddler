#Get all relatively prime pairs with max value of x.
#Returns a list of all pairs.
def get_brood_pairs(x) :
    pairs = []
    #Larger number.
    for b in range(2, x+1):
        #Smaller number.
        for a in range(1, b):
            #Possible divisors.
            for d in range (1, a+1):
                if ((b % d == 0) and (a % d == 0)) :
                    gcd = d
        
            if (gcd == 1):
                pairs.append((a,b))

    return(pairs)


#Determines interferences for two broods of max value x.
#Returns a dataframe of Brood Cycle A, Brood Cycle B, and the Interference.
#Brood Cycles are relatively prime.
def get_interferences(x) :

    #Get brood pairs.
    pairs = get_brood_pairs(x)
    #Initiate lists.
    a_vals = []
    b_vals = []
    interferences = []
    
    for (a,b) in pairs :
        interference = 0
        inter = False
        for m in range(1, b+1) :
            for n in range(1, a+1):
                if a*m + 1 == b*n:
                    interference = b*n
                    inter = True
                    break
                
                elif a*m == b*n + 1:
                    interference = a*m
                    inter = True
                    break
            if inter :
                break
        a_vals.append(a)
        b_vals.append(b)
        interferences.append(interference)

    df = pd.DataFrame(data = {"A": a_vals,
                                    "B": b_vals,
                                    "Max Inter": interferences})
    return(df)
