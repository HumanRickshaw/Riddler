h = 10

#List of number of holds for each simulation run.
num_holds = []
#Run 10,000,000 simulations
for i in range(10000000) :
    
    #Set the heigt, list of all holds, and the maximum distance between any two consecutive holds in the list.
    height = h
    holds = [0, height]
    maximum = height

    #Keep adding holds until the maximum is less than or equal to 1.
    while maximum > 1 :
        #Add a random hold.
        holds.append(height*random.random())        
        holds.sort()
        
        #Determine all distances between two consecutive holds.
        differences = []
        for d in range(1, len(holds)):
            differences.append(holds[d]-holds[d-1])
        
        #Update maximum
        maximum = max(differences)
    
    #Append the final count, not including 0 and the height.
    num_holds.append(len(holds)-2)

#Average of 10,000,000 simulations.
mean = round(statistics.mean(num_holds), 2)

#Plot.
sns.set()
fig = plt.figure(figsize = (14, 8))
ax = fig.add_subplot()

#Histogram.
plt.hist(x = num_holds,
         #Number of bins.
         bins = 211,
         color = '#1F968B')
#Average value vertical line.

            lw = 3,
