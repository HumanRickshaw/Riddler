import math
import matplotlib
from matplotlib.patches import Ellipse as mpE
from matplotlib.patches import Rectangle as mpR
import matplotlib.pyplot as plt

#Plot.
def stuff(ax, alpha):

    #Remove axes and ticks.
    ax.spines["top"].set_visible(False)
    ax.spines["right"].set_visible(False)
    ax.spines["bottom"].set_visible(False)
    ax.spines["left"].set_visible(False)
    ax.set_xticks([])
    ax.set_yticks([])


    r = math.sqrt(13)
    s = math.sqrt(2)

    large = mpR(xy = (-1, 0),
                width = 3,
                height = 3,
                edgecolor = 'indigo',
                ls = '-',
                lw = 3,
                facecolor = 'mediumpurple',
                alpha = alpha)

    small = mpR(xy = (-2, 3),
                width = s,
                height = s,
                edgecolor = 'darkgreen',
                ls = '-',
                lw = 3,
                facecolor = 'lightgreen',
                angle = -135,
                alpha = alpha)

    semi = mpE(xy = (0, 0),
               width = 2*r,
               height = 2*r,
               edgecolor = 'black',
               ls = '-',
               lw = 3,
               facecolor = 'none')


    ax.add_patch(small);
    ax.add_patch(large);
    ax.add_patch(semi);
