import math
import matplotlib
from matplotlib.animation import FuncAnimation as FA
from matplotlib.patches import Polygon as PG
import matplotlib.pyplot as plt
from matplotlib.ticker import FormatStrFormatter as FMF
import seaborn as sns

sns.set()

#Unknown side to maximize area.
As = [math.pi * r / 10000 for r in range(10000)]

#Edge from Angle.
#Volume from Edge.
Us = []
volumes = []
for A in As:
    U = math.sin(A) * math.sqrt(3) / (2 * math.sin((math.pi - A)/ 2))
    Us.append(U)
    vol = (U * math.sqrt(3 - U**2)) / 12
    volumes.append(vol)


#Plot.    
fig = plt.figure(figsize = (13, 8))
ax = fig.add_subplot()

plt.scatter(x = Us, y = volumes, s = 50, c = '#1F968B')
plt.scatter(x = Us[5000], y = volumes[5000], s = 300, c = '#481567')

ax.set_title("Volume of Tetrahedron vs Length of 6th Side", fontsize = 24)

ax.set_xlabel("Length of 6th Side", fontsize = 18)

matplotlib.rc_file_defaults()

fig = plt.figure(figsize = (7.5, 6))
ax = fig.add_subplot(xlim = (-1, 80),
                     ylim = (-1, 64))

ax.set_title("Volume of Tetrahedron With 5 1-inch Sides", fontsize = 24)

#Remove axes and ticks.
ax.spines["top"].set_visible(False)
ax.spines["right"].set_visible(False)
ax.spines["bottom"].set_visible(False)
ax.spines["left"].set_visible(False)
ax.set_xticks([])
ax.set_yticks([])

font = {'size' : 16}

#Nine artists:
#Left face, right face, front face.
#Back left edge, bottom center edge, back right edge.
#Angle text, edge text, volume text.

lf = PG(xy = [(0, 20), (40, 30), (30, 10), (0, 20)],
        facecolor = 'palegreen')

rf = PG(xy = [(0, 20), (40, 30), (30, 10), (0, 20)],
        facecolor = 'palegreen')

ff = PG(xy = [(0, 20), (0, 20), (30, 10), (0, 20)],
        edgecolor = 'forestgreen',
        ls = '-',
        lw = 2,
        facecolor = 'palegreen')

e1 = PG(xy = [(0, 20), (40, 30)],
        edgecolor = 'forestgreen',
        ls = ':',
        lw = 2,
        facecolor = 'None',
        closed = False)

e2 = PG(xy = [(40, 30), (30, 10)],
        edgecolor = 'forestgreen',
        ls = '-',
        lw = 2,
        facecolor = 'None',
        closed = False)

e3 = PG(xy = [(0, 20), (40, 30)],
        edgecolor = 'forestgreen',
        ls = '-',
        lw = 2,
        facecolor = 'None',
        closed = False)

a_text = ax.text(35, 50, '', **font)
u_text = ax.text(35, 45, '', **font)
v_text = ax.text(35, 40, '', **font)


def init():
    """Initialize nine artists."""
    ax.add_patch(lf)
    ax.add_patch(rf)
    ax.add_patch(ff)
    ax.add_patch(e1)
    ax.add_patch(e2)
    ax.add_patch(e3)
    a_text.set_text('')
    u_text.set_text('')
    v_text.set_text('')
    
    return lf, rf, ff, e1, e2, e3, a_text, u_text, v_text

def animate(i):
    
    """(x,y) is the moving vertex that lengthens the 6th side."""
    x = 35 - 35*math.cos(As[i])
    y = 35*math.sin(As[i]) + 20
    
    """Line styles for edges based on perspective."""
    if As[i] < (math.pi - math.atan(2)):
        ls1 = ':'
        ls2 = '-'
        ls3 = '-'
    
    elif As[i] < (math.pi - math.atan(8/15)):
        ls1 = ':'
        ls2 = ':'
        ls3 = ':'
    
    else :
        ls1 = '-'
        ls2 = ':'
        ls3 = '-'
        
    """Update nine artists."""
    rf.set_xy([(x, y), (40, 30), (30, 10), (x, y)])
    ff.set_xy([(0, 20), (x, y), (30, 10), (0, 20)])
    
    e1.set_xy([(0, 20), (40, 30)])
    e1.set_linestyle(ls1)
    e2.set_linestyle(ls2)
    e3.set_xy([(x, y), (40, 30)])
    e3.set_linestyle(ls3)
    
    a_text.set_text("Angle: %.2f°" % (180 * As[i] / math.pi))
    u_text.set_text("6th Side: %.2f inches" % (Us[i]))
    v_text.set_text("Volume: %.5f cubic inches" % volumes[i])

    return rf, ff, e1, e2, e3, a_text, u_text, v_text

#Run animation.
anim = FA(fig, animate, init_func = init, frames = 9999, interval = 2, blit = True)       

#Save animation.
anim.save('2021.05.28 Classic.mp4');
