import matplotlib.pyplot as plt

def menorah(lit) :
    fig = plt.figure(figsize = (12, 1.3))
    ax = fig.add_subplot(111, xlim = (-1,8), ylim = (-1.5, 1.5))

    #Remove axes and labels.
    for side in ['bottom', 'left', 'top', 'right']:
        ax.spines[side].set_visible(False)
    ax.set_xticks([])
    ax.set_yticks([])
    
    #Outline of candle
    plt.plot([x for x in range(8)], [0 for y in range(8)], 'o', color='black', markersize = 45)
    
    #Fire up candles!
    for c in range(8):
        if lit[c] == 1 :
            plt.plot(c, 0, 'o', color = 'red', markersize = 42)
            plt.plot(c, 0, 'o', color = 'orange', markersize = 28)
            plt.plot(c, 0, 'o', color = 'yellow', markersize = 14)
        else :
            plt.plot(c, 0, 'o', color = 'white', markersize = 42)
