#Create Data
data = pd.DataFrame(columns = ['Verses',
                               'Forget',
                               'EV'])

for v in range(1, 10001):
    for f in [x/50000 for x in range(1, 251)]:
        ev = (1/f) * (1 / (1-f)**v - 1) / v
        data = data.append({'Verses' : v,
                            'Forget' : f,
                            'EV' : ev},
                           ignore_index = True)
