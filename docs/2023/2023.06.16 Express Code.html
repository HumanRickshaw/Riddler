#Create DataFrame. Event #, Prob Flips, Prob Rolls, Cum Prob Flips, Cum Prob Rolls.
data = pd.DataFrame(columns = ['n',
                               'PF',
                               'PR',
                               'CF',
                               'CR'])

for n in range(1, 26) :
    
    if n == 1 :
            data = data.append({'n' : 1,
                        'PF' : 0,
                        'PR' : 0,
                        'CF' : 0,
                        'CR' : 0},
                       ignore_index = True)
    elif n == 2 :
            data = data.append({'n' : 2,
                        'PF' : 0,
                        'PR' : 1/6,
                        'CF' : 0,
                        'CR' : 1/6},
                       ignore_index = True)        
    
    else :
        
        #Probability of Coin Flips.
        pf = fib[n-2] / 2**(n-1)
        
        #Probability of Dice Rolls.
        pr = (5**(n-2)) / (6**(n-1))
    
        #Cumalitive Probability of Coin Flips.
        cf = 1 - fib[n+1] / 2**(n-1)
    
        #Cumalitive Probability of Dice Rolls.
        cr = 1 - (5/6)**(n-1)
    
        data = data.append({'n' : n,
                            'PF' : pf,
                            'PR' : pr,
                            'CF' : cf,
                            'CR' : cr},
                           ignore_index = True)

def probCompare(columns, title_prep, ylabel, save):
    
    sns.set()
    plot = sns.lmplot(x = 'n',
                      y = 'Probability',
                      data = df.loc[df.Group.isin(columns)],
                      hue = 'Group',
                      fit_reg = False,
                      legend = False,
                      palette = 'viridis',
                      height = 6,
                      aspect = 2,
                      scatter_kws = {"s": 200})

    #Title setup.
    plt.title('Probability of Winning ' + title_prep + ' a Toss/Roll', fontsize = 24)

    #X-axis setup.
    plt.xlabel('Number of Coin Tosses / Dice Rolls', fontsize = 22)

    #Y-axis setup.
    plt.ylabel(ylabel + ' of Winning', fontsize = 22)

    #Ticks
    plt.tick_params(axis = 'both', which = 'major', labelsize = 18)

    #Legend
    plt.legend(title = "Team",          
               title_fontsize = '22',
               labels = ["Coin Tosses", "Dice Rolls"],
               fontsize = '18',)

    plt.savefig("2023.06.16 Express" + str(save) + ".png", bbox_inches = "tight")
