#Create Data
data = pd.DataFrame(columns = ['Tenths',
                               'Hundreths',
                               'Profit'])

for h in range(0,10) :
    for t in range(0,10) :
        
        if (h == 0 & t ==0) :
            p = 0
        
        else :
            #Dollar value of number is tenths and hundreths.
            d = t/10 + h/100
           
            #Reciprocal.
            r = round(1/d, 2)
            
            #Profit.
            p = 100*d*r - 100
        
        
        data = data.append({'Tenths' : t,
                            'Hundreths' : h,
                            'Profit' : p},
                            ignore_index = True)
