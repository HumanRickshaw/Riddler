#Create DataFrame.
data = pd.DataFrame(columns = ['Monty',
                               'You',
                               'Earning'])
#Find earnings for
#0 ≤ m ≤ 0.5 and
#0 ≤ y ≤ 0.5. 
for y in [i/600 for i in range(0, 301)]:
    for m in [j/600 for j in range(0, 301)]:
        
        e = 520*m*y - 160*m - 200*y + 100
        
        data = data.append({'Monty' : m,
                            'You' : y,
                            'Earning' : e},
                           ignore_index = True)
